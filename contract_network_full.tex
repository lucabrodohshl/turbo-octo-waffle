\documentclass[border=10pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{positioning,arrows.meta,shapes}

\begin{document}

\tikzset{
  node/.style={rectangle, draw=blue!60, fill=blue!5, thick, rounded corners, minimum width=2.5cm, minimum height=0.8cm, align=center, font=\small},
  envNode/.style={rectangle, draw=gray, fill=gray!10, thick, dashed, rounded corners, minimum width=2cm, minimum height=0.8cm, align=center, font=\small},
}

\newcommand{\ConArrow}[3][]{\draw[-{Stealth}, thick, shorten >=2pt, shorten <=2pt] (#2) -- (#3);}

\begin{tikzpicture}[node distance=2cm]

% --- COMPONENT NODES ---
\node[node] (Battery) at (0,0) {Battery};
\node[node] (PowerManager) at (2,2.5) {Power\\Manager};
\node[node] (Motor) at (5,2.5) {Motor};
\node[node] (FlightController) at (7.5,1.25) {Flight\\Controller};
\node[node] (NavigationEstimator) at (5,0) {Navigation\\Estimator};

% --- INTERFACES (edges) ---
% Battery <-> PowerManager
% battery_current, battery_voltage
\ConArrow{Battery}{PowerManager}
% power_mode
\ConArrow{PowerManager}{Battery}

% PowerManager <-> Motor
% motor_current
\ConArrow{Motor}{PowerManager}
% voltage_available
\ConArrow{PowerManager}{Motor}

% Motor -> FlightController
% motor_response_time, motor_thrust
\ConArrow{Motor}{FlightController}

% PowerManager -> FlightController
% power_mode
\ConArrow{PowerManager}{FlightController}

% FlightController <-> NavigationEstimator
% nav_position_error
\ConArrow{NavigationEstimator}{FlightController}
% control_error
\ConArrow{FlightController}{NavigationEstimator}

% Motor -> NavigationEstimator
% motor_current
\ConArrow{Motor}{NavigationEstimator}

% PowerManager -> NavigationEstimator
% power_mode
\ConArrow{PowerManager}{NavigationEstimator}

\end{tikzpicture}
\end{document}
